# $scope

### js code

```js

function myFun() {
    $scope.myData = 'hello';
    console.log('$scope.myData in myFun');
    console.log($scope.myData);
    myService.myQ(fun(data)
    {
        console.log('data')
        console.log(data);
    }
    )
}

myFun();

console.log('$scope.myData after myFun()');
console.log($scope.myData);

```

### html code

```html

{{myData}}

```
